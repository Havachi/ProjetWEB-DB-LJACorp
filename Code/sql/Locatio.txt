CREATE DATABASE 

CREATE TABLE `snows` (
	`IDSnow` INT(11) NOT NULL AUTO_INCREMENT,
	`code` VARCHAR(4) NOT NULL,
	`brand` VARCHAR(20) NOT NULL,
	`model` VARCHAR(30) NOT NULL,
	`snowLength` INT(4) UNSIGNED NOT NULL,
	`qtyAvailable` SMALLINT(6) NOT NULL DEFAULT '0',
	`description` VARCHAR(200) NOT NULL DEFAULT '0',
	`dailyPrice` FLOAT UNSIGNED NOT NULL,
	`photo` VARCHAR(50) NULL DEFAULT NULL,
	`active` TINYINT(4) UNSIGNED NOT NULL DEFAULT '0',
	PRIMARY KEY (`id`),
	UNIQUE INDEX `snow_code` (`code`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=10
;

CREATE TABLE `users` (
	`IDUser` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
	`userEmailAddress` VARCHAR(255) NOT NULL COLLATE 'utf8_unicode_ci',
	`userHashPsw` VARCHAR(255) NOT NULL COLLATE 'utf8_unicode_ci',
	PRIMARY KEY (`id`),
	UNIQUE INDEX `userEmailAddress` (`userEmailAddress`)
)
COLLATE='utf8_unicode_ci'
ENGINE=InnoDB
AUTO_INCREMENT=7
;


CREATE TABLE Locations(
   IDLoc INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
   IDUser INT NOT NULL,
   IDSnow INT NOT NULL,
   DateLoc DATE,
   QtyLoc SMALLINT,
   NbDLoc SMALLINT,
   PRIMARY KEY (`IDLoc`),
   FKUser INT REFERENCES Users(IDUser) NOT NULL,
   FKSnow INT REFERENCES Snows(IDSnow) NOT NULL
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
;
